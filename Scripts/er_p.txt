        protected void DeptView(object sender, EventArgs e)
        {

            string connectionString = ConfigurationManager.AppSettings["LocalDatabase"];
            SqlConnection connection = new SqlConnection(connectionString);
            connection.Open();
            string query = "SELECT * FROM DEPARTMENT";

            using (SqlCommand command = new SqlCommand(query, connection))
            {
                SqlDataReader reader = command.ExecuteReader();
                if (reader.HasRows == true)
                {
                    DataRepeater1.DataSource = reader;
                    DataRepeater1.DataBind();
                }
                else
                {
                    DataRepeater1.DataSource = null;
                    DataRepeater1.DataBind();
                    NoDataDiv1.InnerHtml = "No Records Found!";
                }
                reader.Close();
            }
            connection.Close();
        }

        protected void Delete_Dept(object sender, EventArgs e)
        {
            Button btn = (Button)sender;
            string num = btn.CommandArgument;
            string connectionString = ConfigurationManager.AppSettings["LocalDatabase"];
            SqlConnection connection = new SqlConnection(connectionString);
            connection.Open();
            string query = "DELETE FROM DEPARTMENT WHERE DEPT_ID = " + "'" + num + "'";
            using (SqlCommand command = new SqlCommand(query, connection))
            {
                int rowsAffected = command.ExecuteNonQuery();
                if (rowsAffected > 0)
                {
                    Response.Redirect("ErpIndex.aspx");
                }
                else
                {
                    NoDataDiv1.InnerHtml = "No Data Found!";
                }
            }
            connection.Close();
        }

        protected void Dept_Insert(object sender, EventArgs e)
        {
            string connectionString = ConfigurationManager.AppSettings["LocalDatabase"];

            using (SqlConnection connection = new SqlConnection(connectionString))
            {
                connection.Open();
                string query = "INSERT INTO DEPARTMENT VALUES (@Value1, @Value2)";
                try
                {
                    using (SqlCommand command = new SqlCommand(query, connection))
                    {
                        command.Parameters.AddWithValue("@Value1", Request.Form["DeptId1Txt"]);
                        command.Parameters.AddWithValue("@Value2", Request.Form["DeptNameTxt"]);

                        int rowsAffected = command.ExecuteNonQuery();

                        if (rowsAffected > 0)
                        {
                            Response.Write("<h2 style='background-color: green;'>Registered Successfully</h2>");
                            // Response.Redirect("ErpIndex.aspx");
                        }
                        else
                        {
                            Response.Write("Failed to insert the record.");
                        }
                    }
                }
                catch (Exception e2)
                {
                    string msg = e2.Message;
                    if (msg.Contains("PRIMARY KEY"))
                    {
                        Response.Write("<script>alert('DEPARTMENT ID Already Exists')</script>");
                    }
                    if (msg.Contains("UNIQUE KEY"))
                    {
                        Response.Write("<script>alert('Department Name Already Exists')</script>");
                    }
                }
                finally
                {
                    connection.Close();
                }
            }
        }

        protected void Role_Insert(object sender, EventArgs e)
        {
            string connectionString = ConfigurationManager.AppSettings["LocalDatabase"];

            using (SqlConnection connection = new SqlConnection(connectionString))
            {
                connection.Open();
                string query = "INSERT INTO ROLES VALUES (@Value1, @Value2,@Value3)";
                try
                {

                    using (SqlCommand command = new SqlCommand(query, connection))
                    {
                        command.Parameters.AddWithValue("@Value1", Request.Form["RoleIdTxt"]);
                        command.Parameters.AddWithValue("@Value2", Request.Form["DeptId2Txt"]);
                        command.Parameters.AddWithValue("@Value3", Request.Form["RoleNameTxt"]);

                        int rowsAffected = command.ExecuteNonQuery();

                        if (rowsAffected > 0)
                        {
                            Response.Write("<script>alert('Registered Successfully')</script>");
                            // Response.Redirect("ErpIndex.aspx");
                        }
                        else
                        {
                            Response.Write("Failed to insert the record.");
                        }
                    }
                }
                catch (Exception e2)
                {
                    string msg = e2.Message;
                    if (msg.Contains("PRIMARY KEY"))
                    {
                        Response.Write("<script>alert('Role ID Already Exists')</script>");
                    }
                    if (msg.Contains("FOREIGN KEY"))
                    {
                        Response.Write("<script>alert('Department ID Cannot Found')</script>");
                    }

                    // Response.Write("<script>alert('Department ID Cannot Found')</script>");
                }

            }
        }
